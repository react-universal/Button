# @reactunivseral/button

A universal React Button

```jsx
import React from 'react';
import { View, Text } from 'react-native';
import Button from '@reactuniversal/button';

export default function Page() {
    return (
      <View>
        <Button>
          <Text>Click Me</Text>
        </Button>
      </View>
    );
}`;
```
